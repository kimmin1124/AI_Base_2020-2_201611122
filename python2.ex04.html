A = np.vstack([[4,2],[1,3]])
print("A:")
print(A)
print("\n")

e_values, e_vectors = npl.eig(A)
print(e_values)
print(e_vectors)

#eigen vector u1, u2
u1 = np.vstack(e_vectors[:,0])
u2 = np.vstack(e_vectors[:,1])
print("u1: ", u1)
print("u2: ", u2)

#eigen value lambda1, lambda2
l1, l2 = e_values[0], e_values[1]
print("eigen values: ", l1, l2)
print("\n")

#Check
print("Au1: ", np.dot(A, u1))
print("l1*u1: ", l1*u1)

print("Au2: ", np.dot(A, u2))
print("l2*u2: ", l2*u2)
